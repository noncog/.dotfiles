* Basic Controls
Mostly standard, using vim binds with some improvements.
#+BEGIN_SRC conf
# set mod key
set $mod Mod4

# change focus
bindsym $mod+h focus left
bindsym $mod+j focus down
bindsym $mod+k focus up
bindsym $mod+l focus right

# move window
bindsym $mod+Shift+h move left
bindsym $mod+Shift+j move down
bindsym $mod+Shift+k move up
bindsym $mod+Shift+l move right

# drag floaters key
floating_modifier $mod

# kill window
bindsym $mod+Shift+q kill

# window splitting moved to modes area

# change window layout mode
bindsym $mod+Tab layout tabbed
bindsym $mod+grave layout toggle split

# focus between floating and tiling
bindsym $mod+space focus mode_toggle

# make window float
bindsym $mod+Shift+space floating toggle

# fullscreen
bindsym $mod+minus fullscreen toggle
#+END_SRC

* Workspaces
I only use 3 universal workspaces. Some apps get their own workspace. Some apps live in the scratchpad.
#+BEGIN_SRC conf
# name workspaces
set $ws1 "1"
set $ws2 "2"
set $ws3 "3"

# switch to workspace
bindsym $mod+1 workspace number $ws1
bindsym $mod+2 workspace number $ws2
bindsym $mod+3 workspace number $ws3

# move focused container to workspace
bindsym $mod+Shift+1 move container to workspace number $ws1
bindsym $mod+Shift+2 move container to workspace number $ws2
bindsym $mod+Shift+3 move container to workspace number $ws3

# assign workspaces to monitors
workspace $ws1 output HDMI-1-0 eDP-1
workspace $ws2 output HDMI-1-0 eDP-1
workspace $ws3 output eDP-1 HDMI-1-0
#+END_SRC

* Workspace Apps
These apps are given their own workspaces.

I use a particular format for how my apps are managed.
- =key= = focus
- =shift+key= = spawn
- =control+shift+key= = move to app's workspace

* Modes & Moded Apps
** Resize Mode
#+BEGIN_SRC conf
set $resize_mode Resize: (-), (+), (Shift -), (Shift +)
mode "$resize_mode" {
     bindsym $mod+minus resize shrink width 10 px or 10 ppt
     bindsym $mod+equal resize grow width 10 px or 10 ppt
     bindsym $mod+Shift+minus resize shrink height 10 px or 10 ppt
     bindsym $mod+Shift+equal resize grow height 10 px or 10 ppt

     # back to normal
     bindsym $mod+Escape mode "default"
     bindsym $mod+BackSpace mode "default"
     bindsym $mod+Return mode "default"
     bindsym Escape mode "default"
     bindsym BackSpace mode "default"
     bindsym Return mode "default"
}
bindsym $mod+equal mode "$resize_mode"  
#+END_SRC

** Window Splitting
#+BEGIN_SRC conf
set $split_mode Split: (v)ertically, (s)plit horizontally
mode "$split_mode" {
     bindsym $mod+v split h; mode "default"
     bindsym $mod+s split v; mode "default"

     # back to normal
     bindsym $mod+Escape mode "default"
     bindsym $mod+BackSpace mode "default"
     bindsym $mod+Return mode "default"
     bindsym Escape mode "default"
     bindsym BackSpace mode "default"
     bindsym Return mode "default"
}
bindsym $mod+w mode "$split_mode"
#+END_SRC

** Youtube Mode
Probably the pinnacle of my config...

A mini floating youtube player with sizes and positions for laptop and desktop displays.

*** Requirements
#+BEGIN_SRC conf
# Firefox: about:config > full-screen-api.ignore-widgets true
#+END_SRC

*** Dimensions
**** Default
#+BEGIN_SRC conf
# NOTE: These dimensions are for i3-gaps, they will be different for i3.
set $yt_default_w 720
set $yt_default_h 407
#+END_SRC
**** Desktop    
#+BEGIN_SRC conf
# size
set $yt_d_w 720
set $yt_d_h 407
# location
set $yt_d_x 2910 px
set $yt_d_y 1023 px
#+END_SRC
**** Laptop    
#+BEGIN_SRC conf
# size
set $yt_l_w 636
set $yt_l_h 360
# location
set $yt_l_x 1274 px
set $yt_l_y 710 px
#+END_SRC
**** YouTube URL and Title
#+BEGIN_SRC conf
# set to your unique youtube page. ideally a page you will never visit.
# otherwise your firefox will be turned into the youtuber player...
set $youtube_page https://www.youtube.com/premium
set $youtube_title "YouTube Premium - YouTube — Mozilla Firefox"   
# found using xprop
#+END_SRC
**** Mode
#+BEGIN_SRC conf
for_window [title=$youtube_title] floating enable, resize set $yt_default_w $yt_default_h, move position center
for_window [title=$youtube_title] move scratchpad, sticky enable, mark youtube
	   
set $youtube_mode (Y)outube, (L)aptop, (D)esktop
mode "$youtube_mode" {
     bindsym $mod+y exec --no-startup-id firefox --new-window $youtube_page; mode "default"
     bindsym $mod+l [con_mark="youtube"] focus, resize set $yt_l_w $yt_l_h, move position $yt_l_x $yt_l_y; mode "default"
     bindsym $mod+d [con_mark="youtube"] focus, resize set $yt_d_w $yt_d_h, move position $yt_d_x $yt_d_y; mode "default"
     
     # back to normal
     bindsym $mod+Escape mode "default"
     bindsym $mod+BackSpace mode "default"
     bindsym $mod+Return mode "default"
     bindsym Escape mode "default"
     bindsym BackSpace mode "default"
     bindsym Return mode "default"
}
bindsym $mod+Shift+y mode "$youtube_mode"
bindsym $mod+y [con_mark="youtube"] scratchpad show
#+END_SRC
